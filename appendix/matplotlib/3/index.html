def fob(p1,p2,t):
    '''
    First-order Bessel, ä¸€é˜¶è´å¡žå°”æ›²çº¿ï¼Œ0ä¸ªæŽ§åˆ¶ç‚¹ï¼Œåªæœ‰èµ·ç‚¹å’Œç»ˆç‚¹
    å–å€¼tæ—¶ï¼Œ0<=t<=1ï¼Œæ±‚æ­¤æ—¶çš„è½¨è¿¹ç‚¹
    '''
    #print(p1,p2,t)
    return ((1-t)*p1[0]+t*p2[0],(1-t)*p1[1]+t*p2[1])

def Track_point_gen(*points,accuracy=100):
    '''
    points: è´å¡žå°”æ›²çº¿çš„èµ·ç‚¹ã€å¤šä¸ªæŽ§åˆ¶ç‚¹å’Œç»ˆç‚¹ï¼Œa sequence of tuple point
    accuracy: ç²¾ç»†åº¦ï¼Œa scalar, ç”¨äºŽè®¾ç½®ç»˜åˆ¶çš„è½¨è¿¹æ›²çº¿ä¸Šçš„ç‚¹çš„ä¸ªæ•°ï¼Œé»˜è®¤å€¼100ï¼Œè¯·å…·åä¼ é€’
    è¿”å›žå€¼ï¼ša list of historical tracked points , other process points and t
        è¿”å›žä¸€ä¸ªåˆ—è¡¨ï¼Œåˆ—è¡¨ç¬¬ä¸€é¡¹æ˜¯Besselçš„åŽ†å²è½¨è¿¹ç‚¹å’Œå½“å‰è½¨è¿¹ç‚¹é›†åˆï¼Œåˆ—è¡¨ç¬¬äºŒé¡¹æ˜¯ç»˜åˆ¶å½“å‰è½¨è¿¹ç‚¹çš„è¿‡ç¨‹ç‚¹é›†åˆ
        åˆ—è¡¨ç¬¬ä¸‰é¡¹æ˜¯å½“å‰çš„tå€¼

    backup=points
    accuracy=100 if accuracy==None else int(np.fabs(accuracy))
    history=[points[0]] # åŽ†å²å…¨éƒ¨è½¨è¿¹ç‚¹é›†åˆ
    for t in np.linspace(0,1,accuracy):
        others=[] # ç”¨äºŽç»˜åˆ¶Besselæ›²çº¿çš„å…¶å®ƒè¿‡ç¨‹ç‚¹é›†ï¼Œå³å‡ºå›¾ä¸­éžBesselçš„å…¶å®ƒç‚¹ï¼Œç§°ä¸ºâ€œè¿‡ç¨‹ç‚¹â€
        points=backup
        while len(points)>1:
            temp=[]
            for i,e in enumerate(points[1:],1):
                cur_point=fob(points[i-1],e,t) # current tracked point
                temp.append(cur_point)
            points=temp
            others.append(points)

from matplotlib.collections import PatchCollection

def label(xy, text):
    y = xy[1] - 0.15  # shift y-value for label so that it's below the artist
    plt.text(xy[0], y, text, ha="center", family='sans-serif', size=14)
